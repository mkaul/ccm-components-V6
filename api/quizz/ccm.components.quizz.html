Examples
input: 'checkbox', correct: [ 0, 1 ]              // multiple choice (checkboxes), first and second answer is correct
input: 'checkbox', correct: [ true, true, false ] // multiple choice (checkboxes), first and second answer is correct
input: 'radio',    correct: 0                     // single choice (radio buttons), first answer is correct
input: 'text',     correct: [ "foo", "", "" ]     // text inputs, first text input must be "foo" and the others blank
input: 'number',   correct: [ "", "20", "" ]      // number inputs, second text input must be "20" and the others blank
input: [ 'number', { min: 0, max: 10, value 5 } ] // number inputs, input must be between 0 and 10 and default value is 5
